# [Codeforces Round #732 (Div. 2)](https://codeforces.com/contest/1546)
## A. AquaMoon and Two Arrays
若 a 数组的和与 b 数组的和不同则输出 - 1；
否则用双指针，一个指向，同一位置 a > b 的位置，另一个相反，循环直到两数组相等。
```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int M = 1e4;
typedef pair<int, int> PII;
int a[M], b[M];

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n, sum1 = 0, sum2 = 0;
        cin >> n;
        vector<PII> ans;
        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            sum1 += a[i];
        }
        for (int i = 1; i <= n; i++) {
            cin >> b[i];
            sum2 += b[i];
        }
        if (sum1 != sum2) {
            cout << -1 << endl;
            continue;
        }

        for (int i = 0, j = 0; i <= n && j <= n;) {
            while (i <= n && a[i] <= b[i])
                i++;
            while (j <= n && a[j] >= b[j])
                j++;
            while (a[i] > b[i] && a[j] < b[j]) {
                ans.push_back({i, j});
                a[i]--, a[j]++;
            }
        }

        cout << ans.size() << endl;
        for (int i = 0; i < ans.size(); i++) {
            cout << ans[i].first << " " << ans[i].second << endl;
        }
    }
    return 0;
}
```
